$t1 = 0

$t2 = 1

$t3 = 10

 

loop:

  if ($t2 > $t3) goto end_loop   # bgt $t2, $t3, end_loop

  t1 *= t2                       # mul $t1, $t1, $t2

  t2++                           # addi $t2, $t2, 1

  goto loop                      # j loop

 

end_loop:

  result = $t1                   # sw $t1, result
