Example: Determining String Length

Tuesday, 14 August 2018

9:50 AM

char *string = "...";

Char *s = &string[0];

int length = 0;

while (*s != '\0') {

  length++;

  s++;

}

 

  .data

myString;

  .asciiz "..."

  .globl main

  .text

 

main:

  move $t0, $0     # $t0 will be the length

  la $t1, myString # $t1 points to the beginning of the string

  lb $t3, ($t1)

 

loop: 

  beq $t3, $0, endloop # Go to endloop when $t3 == 0

  addi $t0, $t0, 1     # Add 1 to $t0

  addi $t1, $t1, 1     # Add 1 to $t1

  lb $t3, ($t1)        # Load the new value of $t1 into $t3

  j loop               # Repeat the loop

 

endloop:

  # we done, print it out

  move $a0, $t0

  li $v0, 1 # SYSCALL uses $v0 to know what to do!

  syscall

  j $ra
